\end{center}

\vspace{.5in}
\hline
\vspace{.2in}
Pour chaque paire de matrices ci-dessous : 

\begin{itemize}
\item	Calculer le produit matricielle $AB$.
\item Calculer le determinant de $A$ et de $B$
\item Résoudre le système $AX= Y$
	pour $Y=(1,2,3)$.
\end{itemize}

\hline

\vspace{1in}

\begin{enumerate}
\item $$ A = 
\begin{pmatrix}
-6 & 5 & 6 \\ 
0 & 1 & 2 \\ 
4 & 0 & 2
\end{pmatrix}
,\,
B = 
\begin{pmatrix}
1 & -5 & 2 \\ 
4 & -18 & 6 \\ 
-2 & 10 & -3
\end{pmatrix}
$$
\item $$ A = 
\begin{pmatrix}
1 & -5 & -1 \\ 
-1 & 7 & 2 \\

\end{pmatrix}
,\,
B = 
\begin{pmatrix}
4 & 2 & -3 \\ 
0 & 0 & -1 \\ 
2 & 2 & 2
\end{pmatrix}
$$
\item $$ A = 
\begin{pmatrix}
-2 & 0 & -2 \\

4 & 0 & 2
\end{pmatrix}
,\,
B = 
\begin{pmatrix}

-5 & 2 & -3 \\ 
-2 & 0 & -1
\end{pmatrix}
$$
\item $$ A = 
\begin{pmatrix}
1 & -2 & -2 \\ 
0 & 2 & 2 \\ 
3 & -14 & -13
\end{pmatrix}
,\,
B = 
\begin{pmatrix}
2 & 2 & 0 \\ 
6 & -7 & -2 \\ 
-6 & 8 & 2
\end{pmatrix}
$$
\item $$ A = 
\begin{pmatrix}
-3 & 1 & 3 \\ 
-1 & 1 & -1 \\ 
-2 & 0 & 2
\end{pmatrix}
,\,
B = 
\begin{pmatrix}
1 & -1 & -2 \\ 
2 & 0 & -3 \\ 
1 & -1 & -1
\end{pmatrix}
$$
\item $$ A = 
\begin{pmatrix}
1 & 4 & 4 \\ 
-1 & -2 & -2 \\ 
-2 & -8 & -7
\end{pmatrix}
,\,
B = 
\begin{pmatrix}
-2 & -4 & 0 \\ 
-3 & 1 & -2 \\ 
4 & 0 & 2
\end{pmatrix}
$$
\end{enumerate}
