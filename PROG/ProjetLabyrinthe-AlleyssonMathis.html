Labyrinthes¶

Nous allons nous intresser aux labyrinthes, pour cel nous allons nous concentrer sur plusieurs notions :

  • Nous voulons pouvoir crer des labyrinthes carrs de manire rapide, alatoire et de sorte que le labyrinthe soit faisable mais pas trop simple.
  • Nous voulons pouvoir crer des labyrinthes circulaire (comme les jeux sur les boites de crales).
  • Nous voulons pouvoir rsoudre ces labyrinthes.
  • Nous voulons pouvoir rsoudre un labyrinthe partir d'une photo.

Commenons par le dbut, importons tous les packages dont nous auront besoin tout du long.

Ensuite nous allons casser des murs notre labyrinthe tant qu'il n'est pas connexe.

Pour cel on choisit un mur :

  • si des deux ctes il y a le mme nombre c'est que c'est la mme composante, on le laisse.
  • si par contre les nombres sont differents on le casse et on remplace la composante du plus grand nombre par la valeur du plus petit. Et on rpte cel tant que toutes les cases ne sont pas strictement plus petite que 2.

In [6]:
def still_case_over2(M):
    for i in range(1,len(M)-1,2):
        for j in range(1,len(M[0])-1,2):
            if M[i][j] > 1:
                return True
    return False

def get_values_around_case(i,j,M):
    N = []
    for voisin in [(i-1,j),(i+1,j),(i,j-1),(i,j+1)]:
        u,v = voisin
        if (M[u][v] != 0):
            N.append(M[u][v])
    return sorted(N)

def replace_all_by(old_nb,new_nb,M):
    for i in range(len(M)):
        for j in range(len(M[0])):
            if M[i][j] == old_nb:
                M[i][j] = new_nb

def wall_of_maze(M):
    W = []
    for i in range(1,len(M)-1):
        for j in range(1+(i%2),len(M[0])-1,2):
            W.append((i,j))
    return W

def color_to_maze(M,breaks=True):
    walls = wall_of_maze(M)
    while(still_case_over2(M)):
        r = rd.randint(0, len(walls)-1)
        i,j = walls[r]
        values = get_values_around_case(i,j,M)
        if values[0] != values[-1]:
            replace_all_by(values[-1],values[0],M)
            M[i][j] = values[0]
        walls = walls[:r]+walls[r+1:]
    
    if breaks:
        for k in range(min(len(M),len(M[0]),len(walls)//4)):
            r = rd.randint(0, len(walls)-1)
            i,j = walls[r]
            M[i][j] = 1
            walls = walls[:r]+walls[r+1:]

def travel_maze_for_distance(start,end,N):
    case_to_visit = [[start]]
    distance = 1

    while not(end in case_to_visit[distance-1]):
        distance += 1
        tmp = []
        for coords in case_to_visit[distance-2]:
            i,j = coords
            N[i][j] = distance
            for voisin in [(i-1,j),(i+1,j),(i,j-1),(i,j+1)]:
                a,b = voisin
                if N[a][b] == 1:
                    tmp.append(voisin)
        case_to_visit.append(tmp)
        
travel_maze_for_distance(start,end,N)

def get_chemin(start,end,N):
    way = [end]
    i,j = end
    ## On aura besoin d'une fonction auxiliaire pour se dbarrasser des cases de valeur 1 et 0
    def aux_chemin(a):
        if a[1] == 0 or a[1] == 1:
            return len(N)**5
        else:
            return a[1]
    ##  
    while N[i][j] != 2:
        voisins = [[(i-1,j),N[i-1][j]],[(i+1,j),N[i+1][j]],[(i,j-1),N[i][j-1]],[(i,j+1),N[i][j+1]]]
        next_case = (sorted(voisins,key=aux_chemin))[0][0]
        way.append(next_case)
        i,j = next_case
    
    return (way+[start])

def reg_poly(N,decalage=False):
    if decalage:
        T =  np.linspace(0 ,2*np.pi ,N+1)+np.pi*decalage
    else:
        T =  np.linspace(0 ,2*np.pi ,N+1)
    DD = np.ones((2,N+1))
    DD[0,:] = np.cos(T)
    DD[1,:] = np.sin(T)
    return DD

def get_circle_liaisons(CR):
    nb_anneau = len(CR)
    nb_separation = len(CR[0])
    LI = []
    # On a les murs droits entre 2 anneaux :
    for i in range(nb_anneau):
        for j in range(nb_separation):
            if j == nb_separation - 1:
                if i>0:
                    LI.append(((i,j),(i,0)))
            else:
                LI.append(((i,j),(i,j+1)))
    # Et les murs curbs qui forment les anneaux :
    for j in range(nb_separation):
        for i in range(nb_anneau-1):
            LI.append(((i,j),(i+1,j)))
    return LI

def color_to_maze_circle(CR,breaks=True):
    LI = get_circle_liaisons(CR)
    CLI = copy.deepcopy(LI)

    while(still_case_over1_circle(CR)):
        r = rd.randint(0, len(CLI)-1)
        c1,c2 = CLI[r]
        i1,j1 = c1
        i2,j2 = c2
        
        if CR[i1][j1] != CR[i2][j2]:
            ind = LI.index(CLI[r])
